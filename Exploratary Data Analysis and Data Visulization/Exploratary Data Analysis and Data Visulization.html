price has two outliers, so Ä± apply the Tukey method

Q1 <- quantile(amazon1$Price, 0.25)
Q3 <- quantile(amazon1$Price, 0.75)
IQR <- Q3 - Q1
lower_bound <- Q1 - 1.5 * IQR
upper_bound <- Q3 + 1.5 * IQR

amazon1$Price[amazon1$Price < lower_bound] <- lower_bound
amazon1$Price[amazon1$Price > upper_bound] <- upper_bound

par(mfrow=c(1, 1))
boxplot(amazon1$Price, col = "lightgreen", border = "darkgreen", horizontal = TRUE)
